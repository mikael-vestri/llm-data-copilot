name of the query: count_processed_orders

description: calculates both the number of open renditions and total renditions by each user in each office for each client and legal entity for both system types, Nova and TPT

SELECT 
            can.tax_year,
            COALESCE(mc.name, c.name) AS master_client,
            c.name AS legal_entity,
            ct.name as county,
            COUNT(can.id) AS processed_collector_accounts,
            'nova' AS system_type
        FROM nova_collector_account_number can
        JOIN nova_property p 
            ON p.id = can.property_id
        JOIN nova_client_property cp 
            ON cp.property_id = p.id 
            AND cp.tax_year = can.tax_year
        JOIN nova_client c 
            ON c.id = cp.client_id 
        JOIN nova_client mc 
            ON mc.id = c.master_client_id
        JOIN nova_county ct 
            ON ct.id = p.county_id
        WHERE processed_at IS NOT NULL
          AND p.deleted_at IS NULL
          AND cp.tax_year = 2025
        GROUP BY 
            can.tax_year,
            COALESCE(mc.name, c.name),
            c.name,
            ct.name,
            system_type

        UNION ALL

        SELECT 
            can.taxyear AS tax_year,
            COALESCE(le.clientname, le.legalentityname) AS master_client,
            le.legalentityname AS legal_entity,
            aa.assessorname as county,
            COUNT(can.collectoraccountid) AS processed_collector_accounts,
            'tpt' AS system_type
        FROM tpt_collectoraccounts can
        JOIN tpt_assessoraccount_collectoraccount ac 
            ON ac.collectoraccountid = can.collectoraccountid
        JOIN tpt_assessoraccounts aa 
            ON aa.assessoraccountid = ac.assessoraccountid
        JOIN tpt_assessoraccount_property ap 
            ON ap.assessoraccountid = aa.assessoraccountid
        JOIN tpt_properties p 
            ON p.propertyid = ap.propertyid

            ON le.legalentityid = p.legalentityid

          AND can.taxbillstatus = 'Approved'
          AND can.isactive = TRUE
          AND p.isactive = TRUE
          AND le.isactive = TRUE
          AND aa.isactive = TRUE
        GROUP BY 
            can.taxyear,
            COALESCE(le.clientname, le.legalentityname),
            le.legalentityname,
            aa.assessorname,
            system_type
        ORDER BY 
            processed_collector_accounts desc

query name: exceptions_report

description: combines missing_tax_values and missing_jurisictions to create a report that gathers information of missing tax value and missing jurisdictions by user, county, master cliente for 2025

query:


SELECT
  COALESCE(m.user_id, u.id) AS user_id,
  COALESCE(m.email, u.email) AS email,
  COALESCE(
    m.consultant_name,
    COALESCE(u.first_name || ' ' || u.last_name, 'Not Assigned')
  ) AS consultant_name,
  m.county,
  m.state_code,
  m.master_client,
  m.tax_year,
  m.system_type,
  COALESCE(m.count_missing_tax_value, 0) AS count_missing_tax_value,
  COALESCE(m.count_missing_jurisdictions, 0) AS count_missing_jurisdictions
FROM
  (
    SELECT
      COALESCE(mt.user_id, mj.user_id) AS user_id,
      COALESCE(mt.email, mj.email) AS email,
      COALESCE(mt.consultant_name, mj.consultant_name) AS consultant_name,
      COALESCE(mt.county, mj.county) AS county,
      COALESCE(mt.state_code, mj.state_code) AS state_code,
      COALESCE(mt.master_client, mj.master_client) AS master_client,
      COALESCE(mt.tax_year, mj.tax_year) AS tax_year,
      COALESCE(mt.system_type, mj.system_type) AS system_type,
      COALESCE(mt.count_missing_tax_value, 0) AS count_missing_tax_value,
      COALESCE(mj.count_missing_jurisdictions, 0) AS count_missing_jurisdictions
    FROM
      dev.public.v_missing_tax_value_dd mt
      FULL OUTER JOIN dev.public.v_missing_jurisdictions_dd mj ON mt.user_id = mj.user_id
      AND mt.county = mj.county
      AND mt.state_code = mj.state_code
      AND mt.master_client = mj.master_client
      AND mt.tax_year = mj.tax_year
      AND mt.system_type = mj.system_type
  ) m
  LEFT JOIN main.public.nova_users u ON m.user_id = u.id
WHERE
  u.deleted_at IS NULL
ORDER BY
  count_missing_tax_value DESC

quey name: collector_account_number_calc_view

description: gets the tax rate and tax due by each collector. The calculation of tax due and tax rate comes from the property_jurisdiction_calc_view

query:
SELECT can.id AS collector_account_number_id,
    pv.initial_taxable_value,
    pv.revised_taxable_value,
    nt_round_up_2_decimals(COALESCE(pv.revised_taxable_value, pv.initial_taxable_value)::double precision * (can.split_rate / 100::double precision)) AS collector_account_value,
    tax_data.total_tax_rate,
    tax_data.estimated_tax_due
   FROM collector_account_number can
     JOIN property_value pv ON can.property_id = pv.property_id AND can.tax_year = pv.tax_year
     LEFT JOIN LATERAL ( SELECT sum(pjv.tax_rate) AS total_tax_rate,
            sum(pjv.calc_tax_due) AS estimated_tax_due
           FROM property_jurisdiction_calc_view pjv
          WHERE pjv.collector_account_number_id = can.id) tax_data ON true;



query name: property_jurisdiction_calc_view

description: calculates the tax due amd gathers information about tax values like taxable value and tax rate by each jurisdiction, which is the lowest level of granularity.

query:
 WITH calc_values AS (
         SELECT pj.id AS jurisdiction_id,
            pj.collector_account_number_id,
            ter.rate AS tax_rate,
            ter.disc1,
            ter.disc2,
            ter.disc3,
            ter.disc4,
            COALESCE(pj.final_override, pv.revised_taxable_value::numeric, pv.initial_taxable_value::numeric) AS calc_taxable_value,
            COALESCE(pj.split_rate_override, can.split_rate) AS calc_split_rate,
            pj.tax_override
           FROM property_jurisdiction pj
             JOIN collector_account_number can ON pj.collector_account_number_id = can.id
             JOIN property_value pv ON can.property_id = pv.property_id AND can.tax_year = pv.tax_year
             JOIN tax_entity_rate ter ON pj.tax_entity_id = ter.tax_entity_id AND can.tax_year = ter.tax_year AND ter.deleted_at IS NULL
        )
 SELECT jurisdiction_id,
    tax_rate,
    calc_taxable_value,
    calc_split_rate,
    nt_round_up_2_decimals(calc_taxable_value::double precision * (calc_split_rate / 100::double precision)) AS calc_collector_value,
        CASE
            WHEN tax_override IS NOT NULL THEN tax_override
            ELSE nt_round_2_decimals(calc_taxable_value::double precision * (calc_split_rate / 100::double precision) * (tax_rate / 100::numeric)::double precision)
        END AS calc_tax_due,
    disc1,
    disc2,
    disc3,
    disc4,
    collector_account_number_id
   FROM calc_values;
